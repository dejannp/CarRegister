USE CarRegister;



CREATE TABLE CarManuf (
    Id INT PRIMARY KEY,
    Name VARCHAR(50),
    SerialNum VARCHAR(5),
    Active BIT
);



CREATE TABLE CarModel (
    Id INT PRIMARY KEY,
    CarManufId INT,
    Name VARCHAR(50),
    FuelType VARCHAR(10),
    VIN VARCHAR(13),
    FOREIGN KEY (CarManufId) REFERENCES CarManuf(Id)
);
